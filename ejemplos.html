Ejemplos Avanzados de Procedimientos Almacenados

Ejemplo: Manejo de Inventario

Un procedimiento para actualizar inventario después de una venta:


DELIMITER //
CREATE PROCEDURE actualizarInventario(IN productoID INT, IN cantidadVendida INT)
BEGIN
    DECLARE cantidadActual INT;
    SELECT cantidad INTO cantidadActual FROM Inventario WHERE id = productoID;
    IF cantidadActual >= cantidadVendida THEN
        UPDATE Inventario SET cantidad = cantidadActual - cantidadVendida WHERE id = productoID;
    ELSE
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Stock insuficiente';
    END IF;
END //
DELIMITER ;
